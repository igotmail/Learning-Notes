# 1) Create a for loop that uses modulus to check if a number is evenly divisible by 3, 5, or 3 and 5
for num in range(1, 101):
    if num % 3 == 0 and num % 5 == 0:
        print("FizzBuzz")
    elif num % 3 == 0:
        print("Fizz")
    elif num % 5 == 0:
        print("Buzz")
    else:
        print(num)
