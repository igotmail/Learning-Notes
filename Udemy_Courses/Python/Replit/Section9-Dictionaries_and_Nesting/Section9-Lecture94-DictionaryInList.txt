def add_new_country(country, num_visits, cities):
    # 1.1) Create a new dictionary that contains all of the parameters in the function
    new_country = {
        "country": country,
        "visits": num_visits,
        "cities": cities
    }
    # 1.2) Add the dictionary to the travel_log list
    travel_log.append(new_country)
